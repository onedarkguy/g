import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp();

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Password Strength Checker',
      theme: ThemeData(primarySwatch: Colors.blue),
      home: const PasswordStrengthApp(),
    );
  }
}

class PasswordStrengthApp extends StatefulWidget {
  const PasswordStrengthApp();
  @override
  _PasswordStrengthAppState createState() => _PasswordStrengthAppState();
}

class _PasswordStrengthAppState extends State<PasswordStrengthApp> {
  TextEditingController passwordController = TextEditingController();

  String result = "";

  void checkStrength() {
    String pwd = passwordController.text;

    int score = 0;

    bool hasDigit = false;
    bool hasUpper = false;
    bool hasSpecial = false;

    for (int i = 0; i < pwd.length; i++) {
      String ch = pwd[i];

      // check digit
      if (RegExp(r'[0-9]').hasMatch(ch)) {
        hasDigit = true;
      }

      // check uppercase
      if (RegExp(r'[A-Z]').hasMatch(ch)) {
        hasUpper = true;
      }

      // check special (@, #, $)
      if (ch == '@' || ch == '#' || ch == '\$') {
        hasSpecial = true;
      }
    }

    // conditions
    if (pwd.length > 8) score++;
    if (hasDigit) score++;
    if (hasSpecial) score++;
    if (hasUpper) score++;

    // strength
    if (score <= 1) {
      result = "Weak";
    } else if (score == 2) {
      result = "Normal";
    } else if (score == 3) {
      result = "Strong";
    } else {
      result = "Very Strong";
    }

    setState(() {});
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Password Strength Checker")),
      body: Padding(
        padding: EdgeInsets.all(16),
        child: Column(
          children: [
            TextField(
              controller: passwordController,
              obscureText: true,
              decoration: InputDecoration(labelText: "Enter Password"),
            ),

            SizedBox(height: 20),

            ElevatedButton(
              onPressed: checkStrength,
              child: Text("Check Strength"),
            ),

            SizedBox(height: 20),

            Text("Strength: $result"),
          ],
        ),
      ),
    );
  }
}
