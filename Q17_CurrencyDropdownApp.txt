import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp();

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Currency Converter',
      theme: ThemeData(primarySwatch: Colors.blue),
      home: const CurrencyDropdownApp(),
    );
  }
}

class CurrencyDropdownApp extends StatefulWidget {
  const CurrencyDropdownApp();
  @override
  _CurrencyDropdownAppState createState() => _CurrencyDropdownAppState();
}

class _CurrencyDropdownAppState extends State<CurrencyDropdownApp> {
  TextEditingController inputController = TextEditingController();
  TextEditingController resultController = TextEditingController();

  String selectedConversion = "INR → USD";

  List<String> options = ["INR → USD", "USD → INR", "INR → EUR", "INR → GBP"];

  void convert() {
    double value = double.tryParse(inputController.text) ?? 0;
    double result = 0;

    if (selectedConversion == "INR → USD") {
      result = value * 0.012;
    } else if (selectedConversion == "USD → INR") {
      result = value * 83.0;
    } else if (selectedConversion == "INR → EUR") {
      result = value * 0.011;
    } else if (selectedConversion == "INR → GBP") {
      result = value * 0.0095;
    }

    resultController.text = result.toStringAsFixed(2);
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Currency Converter")),
      body: Padding(
        padding: EdgeInsets.all(16),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            // Input
            TextField(
              controller: inputController,
              keyboardType: TextInputType.number,
              decoration: InputDecoration(labelText: "Enter Amount"),
            ),

            SizedBox(height: 20),

            // Dropdown
            Text("Select Conversion"),
            DropdownButton<String>(
              value: selectedConversion,
              onChanged: (val) {
                setState(() {
                  selectedConversion = val!;
                });
              },
              items: options.map((opt) {
                return DropdownMenuItem(value: opt, child: Text(opt));
              }).toList(),
            ),

            SizedBox(height: 20),

            // Button
            Center(
              child: ElevatedButton(onPressed: convert, child: Text("Convert")),
            ),

            SizedBox(height: 20),

            // Result
            TextField(
              controller: resultController,
              readOnly: true,
              decoration: InputDecoration(labelText: "Result"),
            ),
          ],
        ),
      ),
    );
  }
}
