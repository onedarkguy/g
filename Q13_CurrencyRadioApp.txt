import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp();

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Currency Converter',
      theme: ThemeData(primarySwatch: Colors.blue),
      home: const CurrencyRadioApp(),
    );
  }
}

class CurrencyRadioApp extends StatefulWidget {
  const CurrencyRadioApp();
  @override
  _CurrencyRadioAppState createState() => _CurrencyRadioAppState();
}

class _CurrencyRadioAppState extends State<CurrencyRadioApp> {
  TextEditingController inputController = TextEditingController();
  TextEditingController resultController = TextEditingController();

  String conversionType = "INRtoUSD";

  void convert() {
    double value = double.tryParse(inputController.text) ?? 0;
    double result = 0;

    if (conversionType == "INRtoUSD") {
      result = value * 0.012; // approx rate
    } else if (conversionType == "USDtoINR") {
      result = value * 83.0;
    } else if (conversionType == "INRtoEUR") {
      result = value * 0.011;
    } else if (conversionType == "INRtoGBP") {
      result = value * 0.0095;
    }

    resultController.text = result.toStringAsFixed(2);
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Currency Converter")),
      body: Padding(
        padding: EdgeInsets.all(16),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            // Input
            TextField(
              controller: inputController,
              keyboardType: TextInputType.number,
              decoration: InputDecoration(labelText: "Enter Amount"),
            ),

            SizedBox(height: 20),

            // Radio Buttons
            Text("Select Conversion"),

            Row(
              children: [
                Radio(
                  value: "INRtoUSD",
                  groupValue: conversionType,
                  onChanged: (val) {
                    setState(() => conversionType = val.toString());
                  },
                ),
                Text("INR → USD"),
              ],
            ),

            Row(
              children: [
                Radio(
                  value: "USDtoINR",
                  groupValue: conversionType,
                  onChanged: (val) {
                    setState(() => conversionType = val.toString());
                  },
                ),
                Text("USD → INR"),
              ],
            ),

            Row(
              children: [
                Radio(
                  value: "INRtoEUR",
                  groupValue: conversionType,
                  onChanged: (val) {
                    setState(() => conversionType = val.toString());
                  },
                ),
                Text("INR → EUR"),
              ],
            ),

            Row(
              children: [
                Radio(
                  value: "INRtoGBP",
                  groupValue: conversionType,
                  onChanged: (val) {
                    setState(() => conversionType = val.toString());
                  },
                ),
                Text("INR → GBP"),
              ],
            ),

            SizedBox(height: 20),

            // Button
            Center(
              child: ElevatedButton(onPressed: convert, child: Text("Convert")),
            ),

            SizedBox(height: 20),

            // Result
            TextField(
              controller: resultController,
              readOnly: true,
              decoration: InputDecoration(labelText: "Result"),
            ),
          ],
        ),
      ),
    );
  }
}
