import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp();

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Number Properties',
      theme: ThemeData(primarySwatch: Colors.blue),
      home: const NumberPropsApp(),
    );
  }
}

// ---------------- INPUT SCREEN ----------------
class NumberPropsApp extends StatefulWidget {
  const NumberPropsApp();
  @override
  _NumberPropsAppState createState() => _NumberPropsAppState();
}

class _NumberPropsAppState extends State<NumberPropsApp> {
  TextEditingController numberController = TextEditingController();

  bool prime = false;
  bool palindrome = false;
  bool armstrong = false;
  bool perfect = false;

  bool isPrime(int n) {
    if (n <= 1) return false;
    for (int i = 2; i <= n ~/ 2; i++) {
      if (n % i == 0) return false;
    }
    return true;
  }

  bool isPalindrome(int n) {
    String s = n.toString();
    return s == s.split('').reversed.join();
  }

  bool isArmstrong(int n) {
    int sum = 0, temp = n;
    int digits = n.toString().length;

    while (temp > 0) {
      int d = temp % 10;
      sum += (d * d * d); // simplified (3-digit assumption)
      temp ~/= 10;
    }
    return sum == n;
  }

  bool isPerfect(int n) {
    int sum = 0;
    for (int i = 1; i <= n ~/ 2; i++) {
      if (n % i == 0) sum += i;
    }
    return sum == n && n != 0;
  }

  void process() {
    int num = int.tryParse(numberController.text) ?? 0;

    Map<String, String> results = {};

    if (prime) {
      results["Prime"] = isPrime(num) ? "Yes" : "No";
    }
    if (palindrome) {
      results["Palindrome"] = isPalindrome(num) ? "Yes" : "No";
    }
    if (armstrong) {
      results["Armstrong"] = isArmstrong(num) ? "Yes" : "No";
    }
    if (perfect) {
      results["Perfect"] = isPerfect(num) ? "Yes" : "No";
    }

    Navigator.push(
      context,
      MaterialPageRoute(builder: (_) => ResultScreen(results: results)),
    );
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Number Properties")),
      body: Padding(
        padding: EdgeInsets.all(16),
        child: Column(
          children: [
            TextField(
              controller: numberController,
              keyboardType: TextInputType.number,
              decoration: InputDecoration(labelText: "Enter Number"),
            ),

            CheckboxListTile(
              title: Text("Prime"),
              value: prime,
              onChanged: (val) => setState(() => prime = val!),
            ),

            CheckboxListTile(
              title: Text("Palindrome"),
              value: palindrome,
              onChanged: (val) => setState(() => palindrome = val!),
            ),

            CheckboxListTile(
              title: Text("Armstrong"),
              value: armstrong,
              onChanged: (val) => setState(() => armstrong = val!),
            ),

            CheckboxListTile(
              title: Text("Perfect"),
              value: perfect,
              onChanged: (val) => setState(() => perfect = val!),
            ),

            ElevatedButton(onPressed: process, child: Text("Display")),
          ],
        ),
      ),
    );
  }
}

// ---------------- RESULT SCREEN ----------------
class ResultScreen extends StatelessWidget {
  final Map<String, String> results;

  ResultScreen({required this.results});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Results")),
      body: Padding(
        padding: EdgeInsets.all(16),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            ...results.entries.map((e) {
              return Text("${e.key}: ${e.value}");
            }).toList(),

            SizedBox(height: 20),

            ElevatedButton(
              onPressed: () => Navigator.pop(context),
              child: Text("Back"),
            ),
          ],
        ),
      ),
    );
  }
}
